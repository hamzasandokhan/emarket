
  document.getElementById("executeBtn1").addEventListener("click", () => {
    const textarea1 = document.getElementById("textarea1");
    const textarea2 = document.getElementById("textarea2");
    const textarea3 = document.getElementById("textarea3");
    const fileInput = document.getElementById("fileInput");

    const inputText = textarea1.value;
    const lines = inputText.split("\n");
    const filteredLines = lines.filter(line => line.includes("You can also refer your provider"));
    const filteredText = filteredLines.join("\n");
	const filteredText_ips = filteredLines.join("\n");

    textarea2.value = filteredText;

    // Read the content of the .iim file when selected
    const file = fileInput.files[0];
    const reader = new FileReader();
    reader.onload = function (event) {
      // Set the content of textarea3 with the file content
      textarea3.value = event.target.result;
    };
    reader.readAsText(file);
	
  });
 document.getElementById("executeBtn2").addEventListener("click", () => {
      const textarea2 = document.getElementById("textarea2");
      const textarea3 = document.getElementById("textarea3");
      const textarea5 = document.getElementById("textarea5");
      const textarea4 = document.getElementById("textarea4");

      const ipsTextarea2 = extractIPs(textarea2.value);
      const ipsTextarea3 = extractIPs(textarea3.value);

      let resultTextarea5 = "";
      let resultTextarea4 = "";

      ipsTextarea2.forEach(ip => {
        const regex = new RegExp(`\\b${ip.replace(/\./g, "\\.")}\\b`, "g");
        const errorText = textarea2.value.match(regex);
        const errorTextInTextarea3 = textarea3.value.match(regex);

        if (errorText && errorTextInTextarea3) {
          const updatedErrorText = errorText[0].replace(`[${ip}]`, ip);
          resultTextarea5 += textarea3.value.replace(errorTextInTextarea3[0], updatedErrorText) + "\n";
        } else {
          resultTextarea4 += ip + "\n";
        }
      });

      textarea5.value = resultTextarea5;
      textarea4.value = resultTextarea4;
    });

    function extractIPs(text) {
      const regex = /\b(?:\d{1,3}\.){3}\d{1,3}\b/g;
      return text.match(regex) || [];
    }
